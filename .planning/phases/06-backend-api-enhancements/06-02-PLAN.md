from sqlalchemy.ext.asyncio import create_async_engine, async_sessionmaker, AsyncSession
from sqlalchemy.orm import declarative_base
from typing import AsyncGenerator

DATABASE_URL = "sqlite+aiosqlite:///./news_cache.db"

engine = create_async_engine(DATABASE_URL, echo=False)
async_session_maker = async_sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)

Base = declarative_base()

async def get_db() -> AsyncGenerator[AsyncSession, None]:
    async with async_session_maker() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
        finally:
            await session.close()

async def init_db():
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)
```

from datetime import datetime
from api.db.database import Base

class NewsArticle(Base):
    __tablename__ = "news_articles"
    
    id = Column(Integer, primary_key=True, index=True)
    title_hash = Column(String(64), unique=True, index=True)  # SHA256 of URL+title
    title = Column(String(500))
    source = Column(String(100))
    url = Column(String(1000), index=True)
    published = Column(String(100))
    ticker = Column(String(10), index=True)
    sentiment_score = Column(Float, nullable=True)
    fetched_at = Column(DateTime, default=datetime.utcnow, index=True)
    
    __table_args__ = (
        Index('idx_ticker_fetched', 'ticker', 'fetched_at'),
    )
```

Key points:

- Index on ticker and fetched_at for efficient queries

3. Add helper function for incremental refresh:
   ```python
   async def get_fresh_news(tickers: List[str], db: AsyncSession) -> List[NewsArticle]:
       """Get news - from cache if fresh, otherwise fetch and update."""
       # Check last fetch time
       result = await db.execute(
           select(func.max(NewsArticle.fetched_at)).where(
               NewsArticle.ticker.in_(tickers)
           )
       )
       last_fetch = result.scalar()
       
       should_refresh = (
           last_fetch is None or 
           (datetime.utcnow() - last_fetch).total_seconds() > NEWS_CACHE_TTL_MINUTES * 60
       )
       
       if should_refresh:
           # Fetch new articles from existing news_fetcher
           from src.news_fetcher import fetch_all_news
           new_articles = fetch_all_news(tickers)
           
           # Deduplicate and store
           for article in new_articles.get('headlines', []):
               article_hash = hashlib.sha256(
                   (article.get('link', '') + article.get('title', '')).encode()
               ).hexdigest()
               
               exists = await db.execute(
                   select(NewsArticle).where(NewsArticle.title_hash == article_hash)
               )
               if not exists.first():
                   db.add(NewsArticle(
                       title_hash=article_hash,
                       title=article.get('title', ''),
                       source=article.get('source', ''),
                       url=article.get('link', ''),
                       published=article.get('published', ''),
                       ticker=article.get('ticker', 'SPY')
                   ))
           
           await db.commit()
       
       # Return cached articles
       result = await db.execute(
           select(NewsArticle)
           .where(NewsArticle.ticker.in_(tickers))
           .order_by(NewsArticle.fetched_at.desc())
           .limit(50)
       )
       return result.scalars().all()
   ```

   @app.get("/api/news", response_model=NewsResponse)
   async def get_news(db: AsyncSession = Depends(get_db)):

       articles_db = await get_fresh_news(['SPY', 'VIX', 'AAPL', 'TSLA', 'MSFT'], db)
       
       # Convert to response format
       articles = []

               url=item.url,
               time=item.published
           ))
       
       return NewsResponse(

           total_count=len(articles),

       )
